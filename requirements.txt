# ---- Core (always required) -------------------------------------------------

# Screen capture fallback — cross-platform
mss>=9.0.1

# Image processing and array math (Pillow handles the analysis-frame resize in
# the lean CPU build — OpenCV is only needed for the optional GPU build).
numpy>=1.26.0
Pillow>=10.0.0

# YAML configuration
PyYAML>=6.0.1

# ---- Service layer (REST + WebSocket API) -----------------------------------

# FastAPI app, ASGI server, and data validation for the background service.
fastapi>=0.110.0
uvicorn[standard]>=0.27.0
pydantic>=2.6.0

# Config hot-reload file watching (falls back to mtime polling if absent).
watchdog>=4.0.0

# System-audio loopback capture for the audio-reactive effect (FR-EFF-05).
# Pure-Python (cffi); the effect degrades to a dim base colour if it can't open
# a device. WASAPI loopback on Windows, PulseAudio monitor on Linux.
soundcard>=0.4.3

# Windows Graphics Capture backend (FR-CAP-05). Native (Rust) WGC implementation
# that returns numpy frames and captures hardware-overlay/MPO video DXGI misses.
# Windows-only; the WGC backend degrades to DXGI/MSS if it's absent.
# NOTE: WGC does NOT bypass hardware DRM (HDCP/PlayReady stays black).
windows-capture>=2.0.0 ; sys_platform == "win32"

# ---- Feature dependencies (installed by default; features stay opt-in) ------
#  These power optional features that are OFF by default — installing the
#  package does not enable anything, it just means the feature works the moment
#  the user turns it on (no separate `pip install` step). Kept lightweight and
#  cross-platform; the heavy GPU stack below stays opt-in. All are added to the
#  PyInstaller hidden-imports so frozen installers bundle them too.

# zeroconf — _wled._tcp mDNS auto-discovery for WLED (without it, WLED is still
# found via HTTP subnet probe + manual IP entry).
zeroconf>=0.131.0

# paho-mqtt — MQTT bridge + Home Assistant auto-discovery (Phase C). The bridge
# is a no-op until enabled in settings.
paho-mqtt>=1.6.1

# keyring — OS credential store (Windows Credential Manager / macOS Keychain /
# Linux Secret Service) for secrets: the MQTT broker password and the GitHub
# OAuth token. Without it those fall back to a session-only value.
keyring>=24.0.0

# httpx — async HTTP client for the GitHub integration ("Ambient GitHub
# Awareness"): REST polling + OAuth device flow. The integration is a no-op
# until enabled and connected in the Integrations tab.
httpx>=0.27.0

# ---- Optional — platform display/power monitors -----------------------------
#  macOS (lock/sleep + display reconfig):
#    pip install pyobjc-framework-Cocoa pyobjc-framework-Quartz
#  Linux (logind/screensaver + DRM hotplug):
#    pip install dbus-python PyGObject pyudev

# ---- GPU acceleration (installed by default where supported; never breaks) ---
#  GPU is a bonus, not a requirement: at the 80x45 analysis resolution the CPU
#  path is already near-instant. These are installed by default ONLY on the
#  platforms that have prebuilt CUDA wheels (Windows x64 / Linux x86_64) — the
#  PEP 508 markers below mean pip simply SKIPS them on macOS/ARM (where no wheel
#  exists), so the install never breaks. At runtime the engine probes each
#  backend and silently falls back to NumPy/CPU when no CUDA device is present
#  (see ambilight/gpu.py:_probe_cupy/_probe_torch), so a machine with no NVIDIA
#  GPU installs these fine and just doesn't use them. The lean frozen installer
#  still excludes them (build.py --gpu re-includes) to keep the download small.

# CuPy — CUDA-accelerated NumPy replacement; the real GPU speedup when CUDA 12 is
# present. The cuda12x wheel installs regardless of whether CUDA is present and
# is skipped at runtime if no GPU/CUDA is found. CUDA 11 machine? swap the two
# lines for cupy-cuda11x>=12.3.0 (same markers).
cupy-cuda12x>=13.0.0 ; platform_system == "Windows" and platform_machine == "AMD64"
cupy-cuda12x>=13.0.0 ; platform_system == "Linux" and platform_machine == "x86_64"

# PyTorch — alternative GPU resize backend, intentionally NOT installed by
# default. CuPy (above) is the preferred backend and covers GPU acceleration; on
# Windows the default PyPI torch wheel is CPU-only, so the engine (which only
# uses torch via CUDA) would never use it — 200 MB+ of dead weight. To use torch
# instead, install the CUDA build from https://pytorch.org for your CUDA version.
# torch>=2.2.0

# OpenCV — the CUDA resize backend needs a CUDA-enabled OpenCV build (the PyPI
# wheel is CPU-only). On Windows opencv-python is already a hard dependency below
# (the WGC/DXGI backends need it); a CUDA build is a manual, opt-in replacement.
# opencv-python>=4.9.0.80

# ---- Fast Windows capture backends -------------------------------------------
# WGC (windows-capture, above) is REQUIRED on Windows to capture fullscreen
# games and hardware-overlay video — without it capture falls back to MSS, which
# renders those BLACK. dxcam is the DXGI fallback when WGC is unavailable.

# dxcam — DXGI Desktop Duplication wrapper (Windows only)
dxcam>=0.0.5 ; sys_platform == "win32"

# winsdk — Windows Runtime (WinRT) Python bindings (Windows 10 1903+). Powers the
# Notification Flash feature via UserNotificationListener (and the WGC capture
# backend). Windows-only; both features degrade gracefully if it's absent.
winsdk>=1.0.0b10 ; sys_platform == "win32"

# comtypes — COM interop required by the dxcam DXGI backend (Windows only).
# dxcam drives the display through comtypes-generated COM wrappers; without it
# the DXGI backend fails to import and capture silently degrades to MSS.
comtypes>=1.4.1 ; sys_platform == "win32"

# opencv-python — required by the WGC backend on Windows: windows-capture imports
# cv2 at module load, and dxcam uses it for colour conversion. A hard dependency
# of windows-capture (listed here explicitly so the build always bundles it);
# without it both WGC and DXGI fail to load and capture is stranded on MSS.
opencv-python>=4.9.0 ; sys_platform == "win32"

# pywin32 — Windows API bindings (Windows only)
# pywin32>=306
